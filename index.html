Abilita CORS per tutte le route # Configurazione database DATABASE = 'spese_condivise.db' def init_db(): """Inizializza il database con le tabelle necessarie""" conn = sqlite3.connect(DATABASE) cursor = conn.cursor() # Tabella gruppi cursor.execute(''' CREATE TABLE IF NOT EXISTS groups ( id INTEGER PRIMARY KEY AUTOINCREMENT, code TEXT UNIQUE NOT NULL, name TEXT NOT NULL, description TEXT, created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP ) ''') # Tabella partecipanti cursor.execute(''' CREATE TABLE IF NOT EXISTS participants ( id INTEGER PRIMARY KEY AUTOINCREMENT, group_code TEXT NOT NULL, name TEXT NOT NULL, joined_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP, FOREIGN KEY (group_code) REFERENCES groups (code), UNIQUE(group_code, name) ) ''') # Tabella spese cursor.execute(''' CREATE TABLE IF NOT EXISTS expenses ( id INTEGER PRIMARY KEY AUTOINCREMENT, group_code TEXT NOT NULL, description TEXT NOT NULL, amount REAL NOT NULL, date TEXT NOT NULL, payer TEXT NOT NULL, participants TEXT NOT NULL, -- JSON array dei partecipanti created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP, FOREIGN KEY (group_code) REFERENCES groups (code) ) ''') conn.commit() conn.close() def generate_group_code(): """Genera un codice gruppo univoco""" while True: code = 'SPESE-' + ''.join(random.choices(string.ascii_uppercase + string.digits, k=8)) conn = sqlite3.connect(DATABASE) cursor = conn.cursor() cursor.execute('SELECT code FROM groups WHERE code = ?', (code,)) if not cursor.fetchone(): conn.close() return code conn.close() @app.route('/') def serve_app(): """Serve l'applicazione principale""" return send_from_directory('.', 'index.html') @app.route('/') def serve_static(filename): """Serve file statici""" return send_from_directory('.', filename) # === API GRUPPI === @app.route('/api/groups', methods=['POST']) def create_group(): """Crea un nuovo gruppo""" try: data = request.get_json() name = data.get('name', '').strip() description = data.get('description', '').strip() if not name: return jsonify({'error': 'Nome gruppo richiesto'}), 400 code = generate_group_code() conn = sqlite3.connect(DATABASE) cursor = conn.cursor() cursor.execute( 'INSERT INTO groups (code, name, description) VALUES (?, ?, ?)', (code, name, description) ) conn.commit() conn.close() return jsonify({ 'code': code, 'name': name, 'description': description, 'message': 'Gruppo creato con successo' }), 201 except Exception as e: return jsonify({'error': str(e)}), 500 @app.route('/api/groups/', methods=['GET']) def get_group(code): """Ottieni informazioni su un gruppo""" try: conn = sqlite3.connect(DATABASE) cursor = conn.cursor() # Ottieni info gruppo cursor.execute('SELECT name, description, created_at FROM groups WHERE code = ?', (code,)) group_data = cursor.fetchone() if not group_data: return jsonify({'error': 'Gruppo non trovato'}), 404 # Ottieni partecipanti cursor.execute('SELECT name, joined_at FROM participants WHERE group_code = ?', (code,)) participants = [{'name': row[0], 'joined_at': row[1]} for row in cursor.fetchall()] # Ottieni spese cursor.execute( 'SELECT id, description, amount, date, payer, participants, created_at FROM expenses WHERE group_code = ?', (code,) ) expenses = [] for row in cursor.fetchall(): expenses.append({ 'id': row[0], 'description': row[1], 'amount': row[2], 'date': row[3], 'payer': row[4], 'participants': json.loads(row[5]), 'created_at': row[6] }) conn.close() return jsonify({ 'code': code, 'name': group_data[0], 'description': group_data[1], 'created_at': group_data[2], 'participants': participants, 'expenses': expenses }) except Exception as e: return jsonify({'error': str(e)}), 500 # === API PARTECIPANTI === @app.route('/api/groups//participants', methods=['POST']) def add_participant(code): """Aggiungi un partecipante al gruppo""" try: data = request.get_json() name = data.get('name', '').strip() if not name: return jsonify({'error': 'Nome partecipante richiesto'}), 400 conn = sqlite3.connect(DATABASE) cursor = conn.cursor() # Verifica che il gruppo esista cursor.execute('SELECT code FROM groups WHERE code = ?', (code,)) if not cursor.fetchone(): return jsonify({'error': 'Gruppo non trovato'}), 404 # Aggiungi partecipante try: cursor.execute( 'INSERT INTO participants (group_code, name) VALUES (?, ?)', (code, name) ) conn.commit() except sqlite3.IntegrityError: return jsonify({'error': 'Partecipante già presente nel gruppo'}), 400 conn.close() return jsonify({ 'name': name, 'message': 'Partecipante aggiunto con successo' }), 201 except Exception as e: return jsonify({'error': str(e)}), 500 @app.route('/api/groups//participants/', methods=['DELETE']) def remove_participant(code, name): """Rimuovi un partecipante dal gruppo""" try: conn = sqlite3.connect(DATABASE) cursor = conn.cursor() # Rimuovi partecipante cursor.execute('DELETE FROM participants WHERE group_code = ? AND name = ?', (code, name)) if cursor.rowcount == 0: return jsonify({'error': 'Partecipante non trovato'}), 404 # Rimuovi anche le spese associate cursor.execute('DELETE FROM expenses WHERE group_code = ? AND payer = ?', (code, name)) conn.commit() conn.close() return jsonify({'message': 'Partecipante rimosso con successo'}) except Exception as e: return jsonify({'error': str(e)}), 500 # === API SPESE === @app.route('/api/groups//expenses', methods=['POST']) def add_expense(code): """Aggiungi una spesa al gruppo""" try: data = request.get_json() description = data.get('description', '').strip() amount = data.get('amount') date = data.get('date', '').strip() payer = data.get('payer', '').strip() participants = data.get('participants', []) if not all([description, amount, date, payer, participants]): return jsonify({'error': 'Tutti i campi sono richiesti'}), 400 try: amount = float(amount) except ValueError: return jsonify({'error': 'Importo non valido'}), 400 if amount <= 0: return jsonify({'error': 'Importo deve essere positivo'}), 400 conn = sqlite3.connect(DATABASE) cursor = conn.cursor() # Verifica che il gruppo esista cursor.execute('SELECT code FROM groups WHERE code = ?', (code,)) if not cursor.fetchone(): return jsonify({'error': 'Gruppo non trovato'}), 404 # Verifica che il pagatore sia un partecipante cursor.execute('SELECT name FROM participants WHERE group_code = ? AND name = ?', (code, payer)) if not cursor.fetchone(): return jsonify({'error': 'Il pagatore deve essere un partecipante del gruppo'}), 400 # Verifica che tutti i partecipanti esistano for participant in participants: cursor.execute('SELECT name FROM participants WHERE group_code = ? AND name = ?', (code, participant)) if not cursor.fetchone(): return jsonify({'error': f'Partecipante {participant} non trovato nel gruppo'}), 400 # Aggiungi spesa cursor.execute( 'INSERT INTO expenses (group_code, description, amount, date, payer, participants) VALUES (?, ?, ?, ?, ?, ?)', (code, description, amount, date, payer, json.dumps(participants)) ) expense_id = cursor.lastrowid conn.commit() conn.close() return jsonify({ 'id': expense_id, 'description': description, 'amount': amount, 'date': date, 'payer': payer, 'participants': participants, 'message': 'Spesa aggiunta con successo' }), 201 except Exception as e: return jsonify({'error': str(e)}), 500 @app.route('/api/groups//expenses/', methods=['DELETE']) def remove_expense(code, expense_id): """Rimuovi una spesa dal gruppo""" try: conn = sqlite3.connect(DATABASE) cursor = conn.cursor() cursor.execute('DELETE FROM expenses WHERE id = ? AND group_code = ?', (expense_id, code)) if cursor.rowcount == 0: return jsonify({'error': 'Spesa non trovata'}), 404 conn.commit() conn.close() return jsonify({'message': 'Spesa rimossa con successo'}) except Exception as e: return jsonify({'error': str(e)}), 500 # === API BILANCI === @app.route('/api/groups//balances', methods=['GET']) def get_balances(code): """Calcola e restituisce i bilanci del gruppo""" try: conn = sqlite3.connect(DATABASE) cursor = conn.cursor() # Verifica che il gruppo esista cursor.execute('SELECT code FROM groups WHERE code = ?', (code,)) if not cursor.fetchone(): return jsonify({'error': 'Gruppo non trovato'}), 404 # Ottieni partecipanti cursor.execute('SELECT name FROM participants WHERE group_code = ?', (code,)) participants = [row[0] for row in cursor.fetchall()] # Ottieni spese cursor.execute( 'SELECT amount, payer, participants FROM expenses WHERE group_code = ?', (code,) ) expenses = [] for row in cursor.fetchall(): expenses.append({ 'amount': row[0], 'payer': row[1], 'participants': json.loads(row[2]) }) conn.close() # Calcola bilanci balances = {} for participant in participants: balances[participant] = 0 for expense in expenses: share_per_person = expense['amount'] / len(expense['participants']) # Il pagatore riceve credito balances[expense['payer']] += expense['amount'] # Ogni partecipante deve la sua quota for participant in expense['participants']: balances[participant] -= share_per_person # Calcola rimborsi reimbursements = calculate_reimbursements(balances) return jsonify({ 'balances': balances, 'reimbursements': reimbursements }) except Exception as e: return jsonify({'error': str(e)}), 500 def calculate_reimbursements(balances): """Calcola i rimborsi ottimali""" creditors = [] debtors = [] for person, balance in balances.items(): if balance > 0.01: creditors.append({'person': person, 'amount': balance}) elif balance < -0.01: debtors.append({'person': person, 'amount': -balance}) reimbursements = [] i = j = 0 while i < len(creditors) and j < len(debtors): creditor = creditors[i] debtor = debtors[j] amount = min(creditor['amount'], debtor['amount']) if amount > 0.01: reimbursements.append({ 'from': debtor['person'], 'to': creditor['person'], 'amount': round(amount, 2) }) creditor['amount'] -= amount debtor['amount'] -= amount if creditor['amount'] < 0.01: i += 1 if debtor['amount'] < 0.01: j += 1 return reimbursements # === API UTILITÀ === @app.route('/api/health', methods=['GET']) def health_check(): """Controllo stato del server""" return jsonify({ 'status': 'ok', 'message': 'Server attivo', 'timestamp': datetime.now().isoformat() }) @app.route('/api/stats', methods=['GET']) def get_stats(): """Statistiche generali del sistema""" try: conn = sqlite3.connect(DATABASE) cursor = conn.cursor() cursor.execute('SELECT COUNT(*) FROM groups') total_groups = cursor.fetchone()[0] cursor.execute('SELECT COUNT(*) FROM participants') total_participants = cursor.fetchone()[0] cursor.execute('SELECT COUNT(*) FROM expenses') total_expenses = cursor.fetchone()[0] cursor.execute('SELECT SUM(amount) FROM expenses') total_amount = cursor.fetchone()[0] or 0 conn.close() return jsonify({ 'total_groups': total_groups, 'total_participants': total_participants, 'total_expenses': total_expenses, 'total_amount': round(total_amount, 2) }) except Exception as e: return jsonify({'error': str(e)}), 500 if __name__ == '__main__': #
